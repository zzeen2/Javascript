### 연산자
```js
// 단항 산술 연산자 (+,-.*,/,%) ----------------------------------------------------------
let num = 1;
let num2 =2;
let num3 =num + num2;
console.log(num3);
// output: 3
// 문자열과 숫자를 더하는 경우
// "1" +1 => 문자열 "11"
// 1 + 1 => 숫자 2 
// 형변환
// 연산자를 공부할떄 중요한것은 반환값(쉽게 이해를 돕기 위해서 연산자를 사용한 위치에 생성되는 값)과 데이터 타입 (연산자가 반환하는 데이터타입)

// 비교 연산자 ---------------------------------------------------------
// 두 개 이상의 값을 비교해서 true와 false 값을 반환하는 연산자.
let num4 = 3;
let num5 = 4;
num4 == num5; //is that true? false, data type:bool

// 일반 값을 사용해도 되지만, 이후에 호출해서 사용하지 못하는 휘발성 값
4 == 4 // true

// 문자형 데이터와 숫자형 데이터를 비교하는 경우
"4" == 4; // true, why? 값만 비교하기 때문에 자료형은 신경쓰지 않는다.
// == 데이터 타입 비교 x 값 비교 o > 반환값이 true

// !!!!!데이터 타입까지 비교
// === 
// 데이터 타입과 데이터 값까지 둘다 비교
"4" === 4 // false 값은 같지만 데이터 타입이 다르기 때문에

console.log(num4 == num5);

// 부정 연산자---------------------------------------------------------
// Bool 타입의 값을 반전을 하고 싶을때 반대 되는값이 필요한경우
// 변수의 값을 확인할때
// 사용을 해야하는데 줄일수 있으면 줄였을 경우 가독성이 좀더 좋아진다.

console.log(!true); // true의 값의 반대인 false 
console.log(!false); // flase의 반대인 true

// 비교 연산자와 같이 사용할때
console.log(4 != 4); // 값이 다르니? false

console.log( 4 !== 4); // 값 + 타입까지 다른지 비교

// 큰 값과 작은값을 비교하는 연산자--------------------------------------------
// <, > : 값의 크기를 비교하는 연산자. 초과와 미만을 비교할때
// <=,> =: 이상과 이하를 비교할때 이상과 이하 

```

### 반복문
> 반복적으로 실행되어야하는 코드를 작성할때
> 숫자를 1 2 3 4 5 6 증가시켜서 출력하고싶다

```js
let a = 0;
console.log(++a);
console.log(++a);
console.log(++a);
console.log(++a);
console.log(++a);

// 코드의 간결성, 가독성을 해치는 행위, 확장성
[for 예약어] (선언문; 조건식; 증감식;){
    //스코프 영역 코드 작성 영역
}

// i === index 순서를 표현하기 위한 줄임말
for(let i = 0; i < 10; i++ ){
    console.log(i); //원하는 인덱스만큼 반복되어야할 코드
    // 더이상 실행할 쿠문, 코드 줄이 없으면 
}
for(let i = 0;  i < 10; i++ ){
    for(let j = 0; j<10; j ++){
        console.log(i,j);
    }
}
// let i = 0; 선언문 읽고
// i <10 조건문 확인
// 코드영역을 실행하고
// i++ 증감식 실행
// 조건식 확인
// 코드영역 실행
// 이걸 반복으로 조건문에 충족하지 않을때까지

// 선언문 > 조건식 > 코드실행 > 증감식> 조건문> 코드실행 > 이후 반복
```
